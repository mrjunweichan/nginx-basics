## Setup
  
**Add JWT token to Jump Host (Firefox)**
```sh
cat > /srv/firefox/nginx-repo.jwt <<EOF
<PLACE JWT TOKEN HERE>
EOF
sudo chown ubuntu:ubuntu /srv/firefox/nginx-repo.jwt
```


**Move client cert to Jump Host from (Host Machine)**  
```shell
# Host Machine
scp -O -P 47001 client-cert.p12 ubuntu@xxxx-xxxx-xxxx-xxxx.access.udf.f5.com:/home/ubuntu/
```
  
**Place p12 cert in Jump Host (Firefox)**  
```shell
cp /home/ubuntu/client-cert.p12 /srv/firefox/
sudo chown ubuntu:ubuntu /srv/firefox/client-cert.p12
```
  
**Add hosts in Jump Host VM**  
```shell
sudo vim /etc/hosts

# 10.1.1.6 www.example.com example.api
```
  
**Add Host Entries in Firefox Container**  
```shell
CONTAINER_NAME="firefox"
BACKUP_FILE="backup-${CONTAINER_NAME}-$(date +%Y%m%d_%H%M%S).json"

HOST_ENTRIES=(
  "www.example.com:10.1.1.6"
  "example.api:10.1.1.6"
  "prometheus:10.1.1.4"
  "grafana:10.1.1.4"
)

docker inspect "$CONTAINER_NAME" > "$BACKUP_FILE"

IMAGE=$(jq -r '.[0].Config.Image' "$BACKUP_FILE")
CMD=$(jq -r '.[0].Config.Cmd | join(" ")' "$BACKUP_FILE")
ENV_VARS=$(jq -r '.[0].Config.Env[]?' "$BACKUP_FILE" | awk '{print "-e \"" $0 "\"" }' ORS=' ')
MOUNTS=$(jq -r '.[0].Mounts[]? | "-v \(.Source):\(.Destination)"' "$BACKUP_FILE" | tr '\n' ' ')
PORTS=$(jq -r '.[0].HostConfig.PortBindings // {} | to_entries[] | "-p \(.value[0].HostPort):\(.key | sub("/tcp|/udp"; ""))"' "$BACKUP_FILE" 2>/dev/null | tr '\n' ' ')
PRIVILEGED=$(jq -r '.[0].HostConfig.Privileged' "$BACKUP_FILE")
NETWORK_MODE=$(jq -r '.[0].HostConfig.NetworkMode' "$BACKUP_FILE")

PRIV_FLAG=""
[ "$PRIVILEGED" = "true" ] && PRIV_FLAG="--privileged"

ADD_HOSTS=""
for entry in "${HOST_ENTRIES[@]}"; do
  ADD_HOSTS+="--add-host $entry "
done

docker stop "$CONTAINER_NAME"
docker rm "$CONTAINER_NAME"

CMD_RUN="docker run -d \
  --name \"$CONTAINER_NAME\" \
  --restart=always \
  $PRIV_FLAG \
  $ADD_HOSTS \
  --network $NETWORK_MODE \
  $ENV_VARS \
  $MOUNTS \
  $PORTS \
  $IMAGE $CMD"

eval $CMD_RUN
```
